Installation

Let's discover Docusaurus in less than 5 minutes.

Getting Started​

Get started by creating a new site.

Or try Docusaurus immediately with docusaurus.new.

What you'll need​

  • Node.js version 18.0 or above:
    • When installing Node.js, you are recommended to check all checkboxes related to dependencies.

Generate a new site​

Generate a new Docusaurus site using the classic template.

The classic template will automatically be added to your project after you run the command:

Generell​

mod main;

dec x: i8;
x = 9;
x--;

dec x, y, z: i16 := 0, 5, 10;                                      // declare multiple variables at once

if (x > 5) {                                                       // if x is bigger than 5 then ...
	emitln("'x' (&{x}) is bigger than 5")
} else {                                                          // else ...
	emitln("'x' (&{x}) isn't bigger than 5")
}

for (i: i8 := 0; i <= 8, i++) {                                     // normal for loop
	emitln("i (&{i}) is smaller or equal to 8\n")
}

dec b: i16[x] := [1,5];                                            // creates an array 'b' with a specific specified amount 'x >> 8' of specified types 'i16'

for (i: i8 :: b) {                                                 // for each loop
	emitln("item (&{i}) is in the array named 'b'\n")
}

dec t: f16 := 0.2F;

while (t.isBetween(0, 5)) {                                        // normal while loop (when no expression (b => if true; !b => if false))
	emitln("t = &{t}");
	t++
}

dec z: u8 := (x == 3) ? (x + 1) : (x - 1);                       // inline if statement inside a variable

dec getValueAt(arr: [], i: itype) i16 {
	return arr[i].value                                          // return the value of the element at index 'i' of the given array 'arr'
};

dec getValueError(arr: [], i: u16) ?utype {                       // ? before type means option of type
	if (i < arr.len()) {
		return arr[i].value                                       // return the value of the element at index 'i' of the given array 'arr'
	} else {
		new Error(sEmitf("Cannot find value at index &{i}!", Error.Critical)) // or throw an error
	}
};

dec y: i16 := getValueAt(b, 1);                                    // 'y' will be the 2nd value ('5') of the array 'b'

switch (y) {
	(# < 5, # > 2) => emitln("between 2 and 5"),                  // multiple arguments
	(# < 5) => {                                                  // '??' gets the switch value (in this case 'y')
		emitln("smaller than 5");
		emitln("Just some text")
	},
	(# >= 5) => emitln("bigger or equal to 5"),
	_ => emitln("value is not a number")
}

Lambdas && Imports​

mod lambdas;

uses kvex.Math;     // Imports kvexium math library

// simple lambdas
dec result: i16 := &((x: i8) -> x * x)(4);                // result = 16

// external lambdas
dec &square: i16 := &(x: i16 -> x * x);
dec result: i16 := square(4);                             // result = 16

// complex lambdas
dec &differenceAndAdd: Result{diff: i16, sum: i16} := Lambda((x: i16, y: i16) => {
	if x < y {
		return Result{y-x, x+y}
	} else {
		return Result{x-y, x+y}
	}
});

dec result: i16[2];                                     // an array of 2 i16's
result[0] = differenceAndAdd(9, 7).diff;
result[1] = differenceAndAdd(9, 7).sum;

// or:
result = differenceAndAdd(9, 7);

// named lambdas
dec &namedLambdas: i16 := Dict<str, Lambda>({   // Creates a dictionary with all the lambdas
	"multiply": Lambda((x: i16, y: i16) => x * y),
	"add": Lambda((x: i16, y: i16) => x + y),
	"subtract": Lambda((x: i16, y: i16) => x - y),
	"divide": Lambda((x: i16, y: i16) => x / y)
});
namedFunctions["add"](x, y);                 // accesses the add lambda in the namedFunctions with the parameters x and y provided

Accessmodifiers​

mod accessmodifiers;

dec z: str := "Hello World";  // without modifier -> accessable from everywhere

*dec a: i8 := 2;          // the modifier '*' makes this variable private to this class

&dec b: i32 := 23451;     // the modifier '&' makes this variable final so it cannot change its value
b++;                      // => [ ERROR ] Cannot change value of variable 'b' | [ CAUSE ] Variable is final! | [ SOLUTION ] Create new variable >> assign value of 'b' to it >> change value of new variable!

// Modifiers can also be combined with the '**' which makes the variable protected (private and final at the same time)

**dec c: char := '\0';
c++;                      // => [ ERROR ] Cannot change value of variable 'c' | [ CAUSE ] Variable is protected!

// inside of functions, variables will automaticly be declared as private to this function

Structs, Struct Extending && Interfaces​

mod structs;

struct Person {
	firstName: string | "",   // Default-Wert falls Fehler
	surname: string | "",
	age: u8 | 0
}

struct Student -< Person {    // Student erbt von Person
	grade: u8 | 0,            // neue Eigenschaft
	school: str | "Unknown"
}

dec (person Person) getName() str {   // Methode für Person-Struktur
	return person.firstName + " " + person.surname
}

dec (student Student) getDetails() str {    // Methode für Student-Struktur (erbt auch getName von Person)
	return student.getName() + ", Age: " + student.age + ", Grade: " + student.grade
}

// Verwendung:
dec student: Student := Student{firstName: "John", surname: "Doe", age: 20, grade: 12, school: "Kvexium High School"};
emitln(student.getDetails());    // Output: John Doe, Age: 20, Grade: 12

interface Drawable {                   // provides methods or properties that have to be implemented
	draw()
}

struct Circle -< Drawable {
	radius: f32 | 0.0F,

	dec draw() {
		emitfln("Drawing a circle with radius &{radius}");
	}
}

Print Placeholders​

dec a: i8 := 15;
dec b: str := "abc";
dec c: char := 'x';

emitfln("a = &{%d}", a);

emitfln("a = &{%d}; b = &{%s}", a, b);

emitfln("c = &{%c}", c);

emitfln("a + b = &{%d}", a + b);

emitlfn("a = &{%d}; b = &{%s}; while a = &{%d:1}", a, b); // :1 is just the index where the value is pulled from in this case it is the variable a, with that you can safe space

emitfln("Float Value: &{%f}", floatValue);

emitfln("Hex Value: &{%x}", hexValue);

Annotations​

@NoUse("Don't use due to old version") // Just a warning being automaticly print out that function or variable is old

@Limit(<var>, {<val1>, <val2>}) // simple limit a specific variable
@Limit([{<var1>, {<val1>, <val2>}}, {<var2>, {<val1>, <val2>}}, ...]) // extended limit specific variables

// special @Limit for variable declaration
@Limit({4, 16})
dec a: i8 := 7;

@Just(<var>, [<val1>, <val2>, ...]) // simple limit for the values a variable can have
@Just([{<var1>, [val1, val2, ...]}, {<var2>, [<val1>, <val2>, ...]}, ...]) // simple limit for the values a variable can have

// special @Just for ranges
@Just(f, [1<>10, >=12])             // Just for values where f is between 1 and 10 or f >= 12
dec getNumSquared(f: i8) i32 {
	return f * f
}

@Just(f, <12)             // Just for values where f < 12
dec getNumSquared2(f: i8) i32 {
	return f * f
}

dec result: i32 := getNumSquared(11); // will error because of @Just
dec result2: i32 := getNumSquared2(11); // will pass

@Docs("Just some documentation text") // Used for description of variables, functions or blocks of code :: enshures readability

@Debug // only run function below when debug is enabled, if function called outside of debug throw error
       // will also print every value being inputed and outputed with the function name when debug is run

@Type(<var>, <type>*, <call-function-if-not>) // if a variable matches one or more types
@Type([{<var1>, <type>*, <call-function-if-not1>}, {<var1>, <type>*, <call-function-if-not1>}, ...]) // * just stands for that there can be multiple which then have to be put into [<type>, ...]

You can type this command into Command Prompt, Powershell, Terminal, or any other integrated terminal of your code editor.

The command also installs all necessary dependencies you need to run Docusaurus.

Start your site​

Run the development server:

cd my-website
npm run start

The cd command changes the directory you're working with. In order to work with your newly created Docusaurus site, you'll need to navigate the terminal there.

The npm run start command builds your website locally and serves it through a development server, ready for you to view at http://localhost:3000/.

Open docs/intro.md (this page) and edit some lines: the site reloads automatically and displays your changes. Use plugins to introduce shorter syntax for the most commonly used JSX elements in your project.
